public class HelloWorld {

	public static void main(String[] args) {
		
		class Number {
		
			int number;
			
			public boolean isSquaredNumber() {
				int x = 1;
				int squaredNumber = 1;
				boolean found = false;
				while(squaredNumber < number){
					squaredNumber = x * x;
					
					x+=1;
					
					if(squaredNumber == number) found = true;
				}
				
				return found;
			}

			public boolean isTriangularNumber() {
				int x = 1;
				int trianglarNumber = 1;
				boolean found = false;
				while (trianglarNumber < number){
					trianglarNumber = x*(x+1)/2;
					x+=1;
					
					if (trianglarNumber == number) found = true;
				}
				
				return found;
			}			
		}
		
		Number num = new Number();
		num.number = 25;
		System.out.println(num.isTriangularNumber());	
		System.out.println(num.isSquaredNumber());
	}

}
